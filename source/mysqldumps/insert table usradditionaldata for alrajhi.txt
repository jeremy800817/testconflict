CREATE TABLE `usradditionaldata` (
  `uad_id` bigint NOT NULL AUTO_INCREMENT,
  `uad_userid` bigint NOT NULL,
  `uad_staffid` varchar(50) DEFAULT NULL,
  `uad_status` int NOT NULL DEFAULT '1',
  `uad_createdon` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `uad_createdby` bigint DEFAULT NULL,
  `uad_modifiedon` datetime DEFAULT NULL,
  `uad_modifiedby` bigint DEFAULT NULL,
   PRIMARY KEY (`uad_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
